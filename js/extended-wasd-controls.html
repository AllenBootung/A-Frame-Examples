AFRAME.registerComponent('wasdqerftg-controls', { schema: { fly: {type: 'boolean', default: true} }, init: function() { // register key down/up events // and keep track of all keys currently pressed this.keyPressedSet = new Set(); let self = this; let convertKeyName = function(keyName) { if (keyName == " ") return "Space"; else if (keyName.length == 1) return keyName.toUpperCase(); else return keyName; } document.addEventListener( "keydown", function(eventData) { let keyName = convertKeyName( eventData.key ); // avoid adding duplicates of keys if ( !self.keyPressedSet.has(keyName) ) self.keyPressedSet.add(keyName); } ); document.addEventListener( "keyup", function(eventData) { let keyName = convertKeyName( eventData.key ); self.keyPressedSet.delete(keyName); } ); // movement-related data this.forwardVector = new THREE.Vector3(0,0,-1); this.rightVector = new THREE.Vector3(1,0,0); this.upVector = new THREE.Vector3(0,1,0); // movement/rotation speeds this.moveSpeed = 1; // A-Frame units/second this.turnSpeed = 30; // degrees/second this.lookSpeed = 30; // degrees/second // current rotation amounts this.turnAngle = 0; // around global Y axis this.lookAngle = 0; // around local X axis }, tick: function (time, timeDelta) { // console.log( this.keyPressedSet ); let moveAmount = (timeDelta/1000) * this.moveSpeed; let turnAmount = (timeDelta/1000) * this.turnSpeed * (Math.PI/180); let lookAmount = (timeDelta/1000) * this.lookSpeed * (Math.PI/180); // rotations if (this.keyPressedSet.has("Q")) this.turnAngle += turnAmount; if (this.keyPressedSet.has("E")) this.turnAngle -= turnAmount; if (this.keyPressedSet.has("T")) this.lookAngle += lookAmount; if (this.keyPressedSet.has("G")) this.lookAngle -= lookAmount; this.upVector.set(0,1,0); this.el.object3D.rotation.set(this.lookAngle, 0, 0); this.el.object3D.rotateOnWorldAxis(this.upVector, this.turnAngle); // translations let c = Math.cos(this.turnAngle); let s = Math.sin(this.turnAngle); this.forwardVector.set( -s, 0, -c ).multiplyScalar( moveAmount ); this.rightVector.set( c, 0, -s ).multiplyScalar( moveAmount ); this.upVector.set( 0, 1, 0 ).multiplyScalar( moveAmount ); if (this.keyPressedSet.has("W")) this.el.object3D.position.add( this.forwardVector ); if (this.keyPressedSet.has("A")) this.el.object3D.position.sub( this.rightVector ); if (this.keyPressedSet.has("S")) this.el.object3D.position.sub( this.forwardVector ); if (this.keyPressedSet.has("D")) this.el.object3D.position.add( this.rightVector ); if (this.keyPressedSet.has("R")) this.el.object3D.position.add( this.upVector ); if (this.keyPressedSet.has("F")) this.el.object3D.position.sub( this.upVector ); } });
